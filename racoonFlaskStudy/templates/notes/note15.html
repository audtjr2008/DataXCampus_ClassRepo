# goto News Service
driver.get("https://news.naver.com/main/ranking/popularDay.nhn")
# crawling news article
import re
p = "<dt>\s*?<a\shref=\"(.*?)\".*?>(.*?)<\/a>"
article_p = '<em\sclass=\"img_desc\">(.*?)<a\shref.*?>'
temp = re.findall(p,driver.page_source)

scrapedNews = []
for news in temp:
    print(news[1])
    try:
        driver.get("https://news.naver.com"+news[0].replace("&amp;","&"))
        article = re.findall(article_p ,driver.page_source)
        article = re.sub("<.*?>","",article[0])
        scrapedNews.append({
            'title':news[1],
            'article' :article
        })
    except:
        print("article Loading Fail")

from konlpy.tag import Kkma
from collections import Counter
import nltk
p = r'([ㄱ-하-ㅣ가-힣a-zA-Z0-9]+)+'
k = Kkma()

newsList = []
for i,news in enumerate(scrapedNews):
        newsList.append({
             'title':news['title'],
             'article' :news['article'],
             'tokened_by_Kkma' : Counter(k.nouns(news['article'])).most_common(10),
             'tokened_by_nltk' : Counter(regexp_tokenize(news['article'],p)).most_common(10)
        })

for news in newsList:
    print(news['title'])
    print(news['tokened_by_Kkma'])
    print(news['tokened_by_nltk'])
    print()

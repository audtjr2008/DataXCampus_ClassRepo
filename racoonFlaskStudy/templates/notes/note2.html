def isPerfectNumber1(num):
    div_num = []
    for i in range(1,num):
        if num%i == 0:
            div_num.append(i)
    if sum(div_num) == num:
        return True
    else:
        return False

last_time = time.time()
for i in range(1,10001):
    if isPerfectNumber1(i):
        print(i, 'is Perfect Number')

def formattingList(arg_list):

    if len(arg_list) == 0 :
        items =''
    elif len(arg_list) == 1 :
        items = arg_list[0]
    elif len(arg_list) == 2 :
        items = arg_list[0] + ' and ' +arg_list[1]
    else:
        items=arg_list[0]
        for arg in range(1,len(arg_list)-1):
            items = items +', '+arg_list[arg]
        else:
            items = items + ' and '+arg_list[arg+1]
    return items

def anagram(s1,s2):
    alist = list(s2)

    pos1 = 0
    stillOK = True

    while pos1 < len(s1) and stillOK:
        pos2 = 0
        found = False
        while pos2 < len(alist) and not found:
            if s1[pos1] == alist[pos2]:
                found = True
            else:
                pos2 = pos2 + 1

        if found:
            alist[pos2] = None
        else:
            stillOK = False

        pos1 = pos1 + 1

    return stillOK

input_str1 = input()
input_str2 = input()



if len(input_str1) != len(input_str2):
    print("it is not Anagrams")
else:
    if anagram(input_str1,input_str2):
        print("it is Anagrams")
    else:
        print("it is not Anagrams")

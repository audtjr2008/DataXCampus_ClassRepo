#“what time do i have to be there? what’s the address?”
# “What time do I have to be there? What’s the address?”
# “.”, “!” or “?”.

def capitalizer(arg_str):
    lst = list(arg_str)
    if lst[0].isalpha():
        lst[0] = lst[0].upper()
    for i in range(len(arg_str)):
        if lst[i] in ('.','!','?'):
            for j in range(len(lst[i:])):
                if lst[i+j].isalpha():
                    lst[i+j] = lst[i+j].upper()
                    break
    return ''.join(lst)


input_line = input()
print(capitalizer(input_line))

def is_prime(num):
    for i in range(2,num+1):
        if num%i ==0and num != i:
            return False
    return True

def nextPrime(num):
    cnt_num = num
    while True :
        if is_prime(cnt_num):
            return cnt_num
        else:
            cnt_num += 1

def check_password(arg_pw):
    length_flag = True if len(arg_pw) > 7 else False
    upper_flag = True if any([i.isupper() for i in arg_pw]) else False
    lower_flag = True if any([i.islower() for i in arg_pw]) else False
    number_flag = True if any([i.isnumeric() for i in arg_pw]) else False
    return all([length_flag,upper_flag,lower_flag,number_flag])

input_line = input()
if check_password(input_line):
    print("Your entered password is confirmed")
else:
    print("Please endter right password")
